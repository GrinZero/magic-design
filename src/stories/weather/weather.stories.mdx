 />

export const Template = (args) => {
	const [type,setType]=useState(args.type);
	useEffect(()=>{
		setType(args.type);
	},[args.type]);
	return (
		<div className="mg-w-full flex-center">
			<div className="mg-flex mg-flex-row mg-h-96 mg-items-center mg-justify-around container">
				<Weather {...args} type={type} />
				<Select
					value={type}
					style={{width:"200px"}}
					onChange={(v)=>setType(v)}
					options={[
						{"label":"sunny","value":"sunny"},
						{"label":"cloudy","value":"cloudy"},
						{"label":"rainy","value":"rainy"},
						{"label":"snowy","value":"snowy"},
						{"label":"nighty","value":"nighty"},
					]}
					/>
			</div>
		</div>
	)
};
